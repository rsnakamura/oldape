    def __call__(self, line):

        """
        This only runs if the event is set
        
        :param:

         - `line`: A string to check before propagating
        """
        if not self.event.is_set():
            return
        match = self.regex.search(line)
        if match:
            self.logger.debug("UnFiltered: {0}".format(line))
            self.target(line)
        return
