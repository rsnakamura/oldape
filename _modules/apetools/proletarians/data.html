    @property
    def median(self):
        """
        :return: The median data value
        """
        x = self.data
        if len(x) % TWO == ZERO:
            # n is even, return mean of middle points
            right = len(x)/TWO
            return (x[right - ONE] + x[right]) * HALF
        # else n is odd, return the middle point
        return x[len(x)/TWO]
