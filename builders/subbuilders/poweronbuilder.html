The Power On Parameters¶

class PowerOnParameters(namedtuple("PowerOnParameters",
                                   "identifier switch".split())):
    __slots__ = ()

    def __str__(self):
        return "identifier: {0} switch: {1}".format(self.identifier,
                                                    self.switch)
