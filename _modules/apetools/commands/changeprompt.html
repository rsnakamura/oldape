[docs]        """
        :return: the alternative prompt to use
        """
        if self._prompt is None:
            source = ascii_letters + digits
            self._prompt = (random.choice(ascii_letters) +
                            EMPTY_STRING.join((random.choice(source)) for x in repeat(None,
                                                                                      self.length - 1)))
        return self._prompt
