REVS_PER_DEGREE = 0.002777777777777778
DEGREES_PER_REV = 360.
DEGREES_MIN = 0
DEGREES_MAX = 360


class RotatorConstants(object):
    """
    Holder of Rotator Configuration constants
    """
    __slots__ = ()
    # options
    home_sensor = 'home_sensor'
    home_offset = 'home_offset'
    home_velocity = 'home_velocity'

    # defaults
    default_home_sensor = False
    default_home_offset = -2
    default_home_velocity = 20


class RotatorConfiguration(AMPStepperConfiguration):
    """
    Configuration for the rotator

    :param:

     - `section`: name of section in configuration
     - `configuration`: ConfigParser with section
    """
    def __init__(self, *args, **kwargs):
        super(RotatorConfiguration, self).__init__(*args, **kwargs)
        self._home_sensor = None
        self._home_offset = None
        self._home_velocity = None        
        return

    @property
    def home_velocity(self):
        """
        :return: `home_velocity` value
        :rtype: integer
        """
        if self._home_velocity is None:
            self._home_velocity = self._get_int(RotatorConstants.home_velocity,
                                               optional=True,
                                               default=RotatorConstants.default_home_velocity)
        return self._home_velocity

    @property
    def home_offset(self):
        """
        :return: `home_offset` value
        :rtype: int
        """
        if self._home_offset is None:
            self._home_offset = self._get_int(RotatorConstants.home_offset,
                                             optional=True,
                                             default=RotatorConstants.default_home_offset)
        return self._home_offset

    @property
    def home_sensor(self):
        """
        :return: whether to use a home sensor or not
        :rtype: Boolean
        """
        if self._home_sensor is None:
            self._home_sensor = self._get_boolean(RotatorConstants.home_sensor,
                                                 optional=True,
                                                 default=RotatorConstants.default_home_sensor)
        return self._home_sensor

    def _get_boolean(self, option, optional=False, default=None):
        """
        Gets the option's value as a boolean

        :param:

         - `option`: option in rotator section with desired value
         - `optional`: if True, trap NoOptionErrors
         - `default`: value to return if optional and not found
         
        :raise: NoOptionError if option isn't in section and not optional
        """
        try:
            return self.configuration.getboolean(self.section,
                                                 option)
        except NoOptionError as error:
            if optional:
                return default
            raise
        return


class Rotator(AMPStepperController):
    """
    Turntable controller
    """    
    def __init__(self, *args, **kwargs):
        """
        Rotator constructor

        :param:
        
         - `configuration`: RotatorConfiguration
         - `address`: not sure, output of sending 'DA' command to the device
         - `lock`: re-entrant lock to block serial calls
         - `encoder`: boolean,  if False causes some methods to raise error
         - `debug`: boolean, if True sends print messages to std out
        """
        AMPStepperController.__init__(self, *args, **kwargs)
        self._stepRange = None
        self._serial = None
        self._targetAngle = None
        self._stepsPerDegree = None
        return

    @property
    def stepsPerDegree(self):
        if self._stepsPerDegree is None:
            self._stepsPerDegree = self.configuration.ppr/DEGREES_PER_REV
        return self._stepsPerDegree

    @property
    def targetAngle(self):
        """
        Current angle
        """
        if self._targetAngle is None:
            self._targetAngle = self.currentAngle()
        return self._targetAngle

    @property
    def serial(self):
        """
        serial connection to the device
        """
        if self._serial is None:
            self._serial = super(Rotator, self).serial

            # side-effects
            self._homeSensorEnabled = self.configuration.home_sensor
            if self.configuration.home_sensor:
                self._enableHomeSensor()
            else:
                self._disableHomeSensor()
                                            
            # sendCommand defined in AMPStepperController
            self.sendCommand("EF", 2)   # enable stall prevention with timeout
        return self._serial        
        
    def rotateAbsolute(self, degrees):
        """
        Rotates the table to an angle (0 <= degrees < 360)

        :param:

         - `degrees`: angle to rotate table to

        :raise: Exception if angle is out of bounds
        """
        if DEGREES_MIN <= degrees < DEGREES_MAX:
            self.logger.debug("Rotating to {0} degrees (clockwise)".format(degrees))
            self._targetAngle = degrees
            self.moveAbsoluteSteps(int(degrees*self.stepsPerDegree))
        else:
            raise Exception('Angle out of bounds [{0},{1})'.format(DEGREES_MIN,
                                                                   DEGREES_MAX))
            
            
    def rotateRelative(self, degrees, validate=True):
        """
        Rotate to degrees relative to current angle

        :param:

         - `degrees`: angle to rotate
         - `validate`: if True, raise exception if target angle out of range

        :raise: Exception if `validate` and target out of range
        """
        curAngle = self.currentAngle()
        target = curAngle + degrees
        if validate:
            if DEGREES_MIN <= target < DEGREES_MAX:
                self._targetAngle = target
                self.moveRelativeSteps(int(degrees*self.stepsPerDegree))
            else:
                raise Exception('Angle out of bounds [{0}, {1})'.format(DEGREES_MIN,
                                                                        DEGREES_MAX))
        else:
            self._targetAngle = target
            self.moveRelativeSteps(int(degrees*self.stepsPerDegree))
        return
            
            
    def currentAngle(self):
        """
        Get the current angle

        :return: current angle
        """
        return self.getPositionSteps()/self.stepsPerDegree
                                
    def setAngle(self, degrees):
        """
        Set the table's current angle

        :param:

         - `degrees`: angle to tell the table it's currently at
        """
        self.logger.info("Setting table's current position to {0} degrees".format(degrees))
        self.setPosition(int(round(degrees*self.stepsPerDegree)))
        return
            
    def setMotionProfile(self, vel=None, accel=None, decel=None):
        """
        Sets Angular Velocity, Acceleration, and Decceleration

        :param:

         - `vel`: angular velocity
         - `accel`: angular acceleration
         - `decel`: angular deceleration
        """
        if vel:
            self.logger.debug("Setting angular velocity to {0}".format(vel))
            self.setAngularVelocity(vel)
        if accel:
            self.logger.debug("Setting angular acceleration to {0}".format(accel))
            self.setAngularAccel(accel)
        if decel:
            self.logger.debug("Setting angular deceleration to {0}".format(decel))
            self.setAngularDecel(decel)
        return

        angularVel = round(vel/DEGREES_PER_REV, 7)

        return
    
    def setAngularAccel(self, accel):
    
        """
        Set the angular acceleration

        :param:

         - `accel`: angular acceleration
        """
        angularAccel= round(accel/DEGREES_PER_REV, 7)
        self.setRotationalAccel(angularAccel)
        return        
        
    def setAngularDecel(self, decel):
        """
        Sets the angular deceleration

        :param:

         - `decel`: angular deceleration
        """
        angularDecel = round(decel/DEGREES_PER_REV, 7)
        self.setRotationalDecel(angularDecel)
        return        
        
    def getAngularVelocity(self):
        """
        Converts Revs per second to angular velocity

        :return: angular velocity
        """
        angularVel = self.getRevsPerSec()*DEGREES_PER_REV
        return angularVel
    
    
    def getAngularAccel(self):
        """
        :return: angular acceleration
        """
        angularAccel= self.getRotationalAccel()*DEGREES_PER_REV
        return angularAccel
        
        
    def getMotionProfile(self):
        """
        puts angular velocity, acceleration, deceleration in a dict

        :return: `vel`, `accel`, `decel` dict
        """
        return {'vel':self.getAngularVelocity(),
                'accel':self.getAngularAccel(),
                'decel':self.getAngularDecel()}        
        
        
    def getAngularDecel(self):
        """
        :return: angular deceleration
        """
        angularDecel= self.getRotationalDecel()*DEGREES_PER_REV
        return angularDecel
    
    
    def moveHome(self):
        """
        returns the turntable to angle 0
        """
        if self.configuration.home_sensor:
            #self._setPPR(self._ppr)
            self.setMoveDistanceSteps(-1) # set up jogging to move in negative direction
            vel = self.configuration.home_velocity
            if vel:            
                prevVel = self.getAngularVelocity()
                self.setAngularVelocity(vel)
            self._enableHomeSensor()
            jogSpeed = round(self.configuration.home_velocity*REVS_PER_DEGREE, 2)
            self.sendCommand("JS", jogSpeed)
            self.sendCommand("CJ")

            self.waitForSpeed(0, timeout=20) # timeout=vel/(self._stepRange/float(self._stepsPerMM)) + 1
            self.rotateRelative(180 + self.configuration.home_offset, validate=False)
            time.sleep(1)
            self.waitForSpeed(0, timeout=20)
            self.setAngle(0)

            if vel:
                self.setAngularVelocity(prevVel)
            
            #self.fixPosition()
        else:
            raise Exception('Cannot move home with disabled home sensor')
        return

class MockRotator(rotation_table.BaseClass):
    """
    A class to test the command-line command
    """
    def __init__(self, configuration, lock=None, address=None, *args, **kwargs):
        """
        MockRotator constructor

        :param:

         - `configuration`: RotatorConfiguration
         - `address`: not sure, output of sending 'DA' command to the device
         - `lock`: re-entrant lock to block serial calls
         - `encoder`: boolean,  if False causes some methods to raise error
         - `debug`: boolean, if True sends print messages to std out
        """
        super(MockRotator, self).__init__()
        self.logger.info(CREATION.format(thing=self.__class__.__name__))
        values = "\n\tconfiguration={0}\n\tlock={1}\n\taddress={0}".format(configuration,
                                                                   lock,
                                                                   address)
        self.logger.info(CALLED_WITH.format(attribute='MockRotator.__init__',
                                            values=values))

        if args:
            self.logger.warning(ARGS.format(value=args))
        if kwargs:
            self.logger.warning(KWARGS.format(value=kwargs))
        self._valid_names = None        
        return

    @property
    def valid_names(self):
        """
        List of non-magic names in the Rotator
        """
        if self._valid_names is None:
            self._valid_names = [name for name in dir(Rotator) if not name.startswith('__')]
        return self._valid_names
        
    def setAngle(self, degrees):
        """
        tell the table what angle it's at

        :param:

         - `degrees`: angle to tell the table it's currently at

        :raise: Exception if degrees out of range
        """
        self.logger.info(CALLED_WITH.format(attribute="MockRotator.setAngle",
                                            values="degrees='{0}'".format(degrees)))
        if not (DEGREES_MIN <= degrees < DEGREES_MAX):
            raise Exception("degrees={0} out of bounds ({1} <= degrees < {2})".format(degrees,
                                                                                      DEGREES_MIN,
                                                                                      DEGREES_MAX))
        return

    def setMotionProfile(self, vel=None, accel=None, decel=None):
        """
        Sets Angular Velocity, Acceleration, and Decceleration

        :param:

         - `vel`: angular velocity
         - `accel`: angular acceleration
         - `decel`: angular deceleration
        """
        self.logger.info(CALLED_WITH.format(attribute='MockRotator.setMotionProfile',
                                            values='vel={0}, accel={1}, decel={2}'.format(vel,
                                                                                          accel,
                                                                                          decel)))
        if vel is not None and not (MIN_VELOCITY <= vel <= MAX_VELOCITY):
            raise Exception("vel out of range ({0})".format(vel))
        if accel is not None and not (MIN_ACCELERATION <= accel <= MAX_ACCELERATION):
            raise Exception("accel out of range ({0})".format(accel))

        if decel is not None and not (MIN_ACCELERATION <= decel <= MAX_ACCELERATION):
            raise Exception("decel out of range ({0})".format(decel))
        return

    def rotateAbsolute(self, degrees):
        """
        Rotates the table to an angle (0 <= degrees < 360)

        :param:

         - `degrees`: angle to rotate table to

        :raise: Exception if angle is out of bounds
        """
        self.logger.info(CALLED_WITH.format(attribute='MockRotator.rotateAbsolute',
                                            values='degrees={0}'.format(degrees)))

        if not (DEGREES_MIN <= degrees < DEGREES_MAX):
            raise Exception("{0} degrees is out of range".format(degrees))
        return

    def waitForPosition(self, timeout=10):
        """
        waits until target position (or timeout) is reached

        :param:

         - `timeout`: seconds to wait before giving up
        """
        self.logger.info(CALLED_WITH.format(attribute='MockRotator.waitForPosition',
                                            values='timeout={0}'.format(timeout)))
        if not type(timeout) in (FloatType, IntType) or timeout <0:
            raise Exception("Invalid timeout -- {0} seconds".format(timeout))
        return

    def __getattr__(self, name, *args, **kwargs):
        """
        A cleanup to get the stuff not implemented
        """
        if name in self.valid_names:
            self.logger.warning("unexpected method call: '{0}'".format(name))
        elif name not in dir(Rotator):
            raise Exception("Unknown Rotator attribute: '{0}'".format(name))
        return
