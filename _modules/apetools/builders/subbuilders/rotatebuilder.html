        if self._parameters is None:
            parameters = []
            values = self.config_map.get_list(ConfigOptions.rotate_section,
                                                    ConfigOptions.angles_option)
            for list_index, item in enumerate(values):
                if COLON in item:
                    angle, velocity = item.split(COLON)
                else:
                    angle, velocity = item, 0

                # check the direction
                clockwise = angle.startswith('-') 
                parameters.append(RotateParameters(angle.lstrip('-'), velocity,
                                                   clockwise))

            self.previous_parameters.append(Parameters(name=RotateBuilderEnums.angle_velocity,
                                                       parameters=parameters))
            self._parameters = self.previous_parameters
        return self._parameters

    def parameters(self):

[docs]        """
        for each angle, create a dict of <table-name:arguments string>
        
        :return: list of named-tuple Parameters
        """
        if self._parameters is None:
            table_parameters = []

            # if any list of angles is shorter than others, then repeat last angle
            # until longest list of angles is done
            # assumes that the RotateArgumentsGenerator operators are overloaded using angle count
            most_angles = max(self.table_arguments_generator.itervalues())

            # use the arguments generator with the most angles as the limit
    
            for angle in xrange(len(most_angles)):
                rotate_params = {}
                
                for table, params in self.table_arguments_generator.iteritems():
                    try:
                        rotate_params[table] = params.argument_strings[angle]
                    except IndexError as error:
                        self.logger.debug(error)
                        self.logger.info("{0} section had fewer angles than {1}, repeating last angle".format(table,
                                                                                                              most_angles.section))
                        rotate_params[table] = params.argument_strings[-1]
                table_parameters.append(rotate_params)
            self.previous_parameters.append(Parameters(name=RotateUsurperConstants.name,
                                                       parameters=table_parameters))
            self._parameters = self.previous_parameters
        return self._parameters

    def get_ssh_parameter(self, table):


[docs]        """
        Convenience method to get ssh parameters

        :rtype: SSHParameters
        """
        constants = RotateUsurperConstants
        hostname = self.config_map.get(table,
                                        constants.hostname_option)
        username = self.config_map.get(table,
                                        constants.username_option)
        password = self.config_map.get(table,
                                        constants.password_option,
                                        optional=True)

        return SSHParameters(hostname=hostname,
                             username=username,
                             password=password,
                             identifier=table)
